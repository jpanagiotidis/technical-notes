# Install (Ubuntu 14.04)
	sudo add-apt-repository ppa:chris-lea/node.js
	sudo apt-get update
	sudo apt-get install nodejs

	## To use npm install -g without sudo (yet-another-option.sh from https://gist.github.com/isaacs/579814#file-yet-another-option-sh)
		echo prefix = ~/.node >> ~/.npmrc
		edit ~/.bashrc and add:
			export PATH="$PATH:$HOME/.node/bin"
		** if something is completely broken and you want to start from scratch, all you need to do is remove your ~/.node directory
